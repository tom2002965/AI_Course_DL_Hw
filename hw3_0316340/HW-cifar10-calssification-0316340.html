  • 可以優先考慮使用CNN
  • 對於沒有GPU環境，請先嘗試較為簡單的模型，不然計算時間會太久

class Net(nn.Module):
    def __init__(self):
        super(Net, self).__init__()
        self.conv1 = nn.Conv2d(3, 6, 5)
        self.pool = nn.MaxPool2d(2, 2)
        self.conv2 = nn.Conv2d(6, 16, 5)
        self.fc1 = nn.Linear(16 * 5 * 5, 120)
        self.fc2 = nn.Linear(120, 84)
        self.fc3 = nn.Linear(84, 10)
        
    def forward(self,x):
        x = self.pool(F.relu(self.conv1(x)))
        x = self.pool(F.relu(self.conv2(x)))
        x = x.view(-1, 16 * 5 * 5)
        x = F.relu(self.fc1(x))
        x = F.relu(self.fc2(x))
        x = self.fc3(x)
        return x

訓練神經網路¶

  • n_epochs 代表要把整個訓練資料循環過幾次
  • train_loss_history 用來保存每個epoch的平均 batch_train_loss
In [203]:
%%time
n_epochs = 5 #you can change here
train_loss_history = []

for epoch in range(n_epochs):
    batch_train_loss_ = []
    
    for _, (data, target) in enumerate(train_loader):
        
        # wrap them in Variable
        inputs, labels = Variable(data), Variable(target)

        # zero the parameter gradients
        train_op.zero_grad()

        # forward + backward + optimize
        outputs = net(inputs)
        batch_train_loss = F.cross_entropy(outputs, labels)
        batch_train_loss.backward()
        train_op.step()
        
        batch_train_loss_.append(batch_train_loss.data[0])
    
    batch_train_loss = np.mean(batch_train_loss_)
    train_loss_history.append(batch_train_loss)
    print ("epoch:%s , train_loss:%s" % (epoch , batch_train_loss))
