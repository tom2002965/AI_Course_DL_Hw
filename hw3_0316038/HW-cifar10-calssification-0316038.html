Q1 定義神經網路模型(40/100)¶

(1)請根據cifar10資料特性設計神經網路架構¶

In [8]:
class Net(nn.Module):
    def __init__(self):
        super(Net, self).__init__()
        
        self.conv1 = nn.Conv2d(3, 6, 5)
        self.conv2 = nn.Conv2d(6, 16, 5)
        self.fc1 = nn.Linear(400,120)
        self.fc2 = nn.Linear(120,84)
        self.fc3 = nn.Linear(84,10)
        
    def forward(self,inputs):
        x = F.relu(self.conv1(inputs))
        x = F.max_pool2d(x, 2)
        x = F.relu(self.conv2(x))
        x = F.max_pool2d(x, 2)
        x = x.view(x.size(0), -1)
        x = F.relu(self.fc1(x))
        x = F.relu(self.fc2(x))
        logits = self.fc3(x)
        
        return logits

%%time
n_epochs = 12 #you can change here
train_loss_history = []

for epoch in range(n_epochs):
    batch_train_loss_ = []
    
    for _, (data, target) in enumerate(train_loader):
        
        data = Variable(data) 
        target = Variable(target)
        
        train_op.zero_grad()

        outputs = net(data)
        batch_train_loss = criterion(outputs, target)
        batch_train_loss.backward()
        train_op.step()
        
        batch_train_loss_.append(batch_train_loss.data[0])
    
    batch_train_loss = np.mean(batch_train_loss_)
    train_loss_history.append(batch_train_loss)
    print "epoch:%s , train_loss:%s" % (epoch , batch_train_loss)
